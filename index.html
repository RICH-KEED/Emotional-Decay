<?php
// PHP variables
$student_name = "Riyana";
$course = "Building Web Applications in PHP";

// PHP array
$programming_languages = array("PHP", "JavaScript", "Python", "Java", "C++");

// PHP function
function calculateGPA($grades) {
    $total = 0;
    $count = count($grades);
    
    foreach ($grades as $grade) {
        $total += $grade;
    }
    
    return round($total / $count, 2);
}

// PHP output
echo "Student: " . $student_name;
echo "Course: " . $course;
?>
